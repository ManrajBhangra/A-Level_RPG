260  def receive_message(s, Self, P):
261      ##### receive_message #######
262      # Parameters :- s:server, self:Knight or Mage, P:other character
263      # Return Type :- none
264      # Purpose :- Allows the server to interact with the clients and has the client receive messages
265      # from the server. These messages change variables within the knight and mage class which allows
266      # the game to produce the actions completed from the other client. It also allows the second client
267      # to know the enemy moves that were calculated on player one's client.
268      ###########################
269      while True:
270          data = s.recv(1024)
271          if not data:
272              continue
273          else:
274              msg = json.loads(data.decode())
275  
276          if msg["Data"]["Move"] == "Attack":  # This 'if' statement reads the message from the server and updates
277              # variables to allow for the moves to happen on the client
278              P.Att = True
279  
280          elif msg["Data"]["Move"] == "En":
281              if msg["Data"]["Target"] == 1:
282                  Self.En_Tar = 1
283              elif msg["Data"]["Target"] == 2:
284                  Self.En_Tar = 2
285  
286              if msg["Data"]["Attack"] == "Attack":
287                  Self.En_Atts = 1
288              elif msg["Data"]["Attack"] == "Special":
289                  Self.En_Atts = 2
290              elif msg["Data"]["Attack"] == "Block":
291                  Self.En_Atts = 3
292  
293          elif msg["Data"]["Move"] == "SP":
294              if msg["Data"]["SP"] == 1:
295                  P.SP1 = True
296              elif msg["Data"]["SP"] == 2:
297                  P.SP2 = True
298              elif msg["Data"]["SP"] == 3:
299                  P.SP3 = True
300              elif msg["Data"]["SP"] == 4:
301                  P.SP4 = True
302  
303          elif msg["Data"]["Move"] == "Block":
304              P.Bl = True
305  
306          elif msg["Data"]["Move"] == "Close":
307              P.S_Cl = True

642  def AI_Attack(HP, Enemies, index, Target, Attacker):
643      ##### AI_Attack #######
644      # Parameters :- HP: integer, Enemies: list of enemies, index: integer, Target: Knight or Mage, Attacker: Enemy
645      # Return Type :- Boolean
646      # Purpose :- More complex part of the enemy AI which will discover if one of your characters can be killed
647      # this turn and returns True if not but False if not meaning the function is only run once and can be called
648      # upon by itself
649      ###########################
650      if len(Enemies.Enemies) - index - 1 >= 1:  # Creates a stopping condition for the recursive function
651          for i in (0, len(Enemies.Enemies) - index - 1):
652              Attacky = Enemies.Enemies[i]
653              if HP - (Attacky.attack * 100 // (100 + Players.get(Target).defense)) <= 0:
654                  # 'if' and 'elif' check to see if character is killable
655                  Attack(Attacker, Players.get(Target))
656                  if GameType == "Multi":
657                      P1.Server_En(s, playerNum, Target, "Attack")
658                  return True
659              elif Attacky.SpGauge - Attacky.SpMoves.get("Cost") >= 0:
660                  if HP - int((Attacky.SpMoves.get("Damage") // 5) * Attacky.attack * 100 // (
661                          100 + Players.get(Target).defense)) <= 0:
662                      Attack(Attacker, Players.get(Target))
663                      if GameType == "Multi":
664                          P1.Server_En(s, playerNum, Target, "Attack")
665                  return True
666              else:
667                  # 'else' happens if not killable with current amount of characters and will check if one more killable
668                  HP -= Attacky.attack * 100 // (100 + Players.get(Target).defense)
669                  index -= 1
670                  AI_Attack(HP, Enemies, index, Target, Attacker)
671      else:
672          return False
